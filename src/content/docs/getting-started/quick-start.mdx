import { Steps, Aside, Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

<Aside type="note" title="Prerequisites">
Before getting started, make sure you have the following installed:

- [**Node.js**](https://nodejs.org/): Version >= 24
- [**Bun**](https://bun.com/): Version >= 1.3
</Aside>

<Steps>

1. Initialize a new PageZERO project:
   <Tabs>
      <TabItem label="Open-source edition">
      ```sh
      bunx pagezero@latest init
      ```
      </TabItem>
      <TabItem label="Paid edition">
      ```sh
      bunx pagezero@latest init --powerup
      ```
        <LinkButton href="https://pagezero.dev/#pricing" variant="secondary" icon="external">
          Learn about PowerUP
        </LinkButton>
      </TabItem>
   </Tabs>

2. Enter your project directory:
   ```sh
   cd <your-project-name>
   ```

3. Run development server:
   ```sh
   bun dev
   ```

4. Open your browser and navigate to `http://localhost:3000` to see the app in action.

</Steps>
