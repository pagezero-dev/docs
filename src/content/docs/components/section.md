## Import

```js
import { Section } from '@/content/components/section'
```

## Usage

```js
<Section
  id="features"
  title="Our Features"
  description="Discover what makes our platform special"
  spacingSize="default"
  background="light"
>
  {/* Your section content here */}
</Section>
```

## Usage without title

```js
<Section spacingSize="lg" background="none">
  {/* Your custom section content here */}
</Section>
```
